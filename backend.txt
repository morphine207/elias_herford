1. GET /api/course/{course_id}

Returns course content (slides + quiz).

2. GET /api/news

Returns news filtered by keyword/topic.

@app.get("/api/news")
def get_news(topic: str):
    with open("data/news.json") as f:
        all_news = json.load(f)
    return [n for n in all_news if topic.lower() in [k.lower() for k in n["keywords"]]]

3. POST /api/feedback

Appends feedback to a file.

@app.post("/api/feedback")
def save_feedback(feedback: dict):
    feedback["timestamp"] = datetime.utcnow().isoformat()
    with open("data/feedback.json", "a") as f:
        f.write(json.dumps(feedback) + "\n")
    return {"status": "received"}

app.mount("/static", StaticFiles(directory="static"), name="static")

âš™ï¸ FastAPI Example App (main.py)
from fastapi import FastAPI
from fastapi.staticfiles import StaticFiles
from fastapi.middleware.cors import CORSMiddleware
import json
from datetime import datetime

app = FastAPI()
app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_methods=["*"],
    allow_headers=["*"],
)

app.mount("/static", StaticFiles(directory="static"), name="static")

@app.get("/api/course/{course_id}")
def get_course(course_id: int):
    with open(f"data/course_{course_id}.json") as f:
        return json.load(f)

@app.get("/api/news")
def get_news(topic: str):
    with open("data/news.json") as f:
        all_news = json.load(f)
    return [n for n in all_news if topic.lower() in [k.lower() for k in n["keywords"]]]

@app.post("/api/feedback")
def save_feedback(feedback: dict):
    feedback["timestamp"] = datetime.utcnow().isoformat()
    with open("data/feedback.json", "a") as f:
        f.write(json.dumps(feedback) + "\n")
    return {"status": "received"}
